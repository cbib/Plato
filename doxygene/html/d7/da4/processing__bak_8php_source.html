829 function processEnzyme(){
830  /* Needed vars... */
831  var slopeRatio = 1.0;
832  var blankRawDataMap = rawDataMap.slice(0, 4);
833  var blankDataTypeMap = dataTypeMap.slice(0, 4);
834  var blankExcludMap = excludMap.slice(0, 4);
835  var blankFwMap = fwMap.slice(0, 4);
836  var maxRawDataMap = rawDataMap.slice(4, 8);
837  var maxDataTypeMap = dataTypeMap.slice(4, 8);
838  var maxExcludMap = excludMap.slice(4, 8);
839  var maxFwMap = fwMap.slice(4, 8);
840  var avgBlankBuffer = bufferAvg(blankRawDataMap, blankExcludMap, blankDataTypeMap);
841  var avgMaxBuffer = bufferAvg(maxRawDataMap, maxExcludMap, maxDataTypeMap);
842 
843  /* Corrected_MAX_Control = AverageOf((MAX_control - AverageOf(MAX_buffer))/freshweight/SloapRatio) */
844  var corrMaxControl = corrControlMaxAverage (maxRawDataMap, maxExcludMap, maxDataTypeMap, maxFwMap, avgMaxBuffer, slopeRatio);
845  /* Corrected_BLANK_Control = AverageOf((BLANK_control - AverageOf(BLANK_buffer))/freshweight) */
846  var corrBlankControl = corrControlBlankAverage (blankRawDataMap, blankExcludMap, blankDataTypeMap, blankFwMap, avgBlankBuffer);
847 
848  for(var i=0;i<4; i++){
849  for(var j=0;j<12; j++){
850  if((excludMap[i][j]==0) && (dataTypeMap[i][j]!=1)) {
851  /*Corrected_BLANK_Val = (rawBLANK-AverageOf(BLANK_buffer)) / frehsweight */
852  var corrBlankValue = blankRawDataMap[i][j]-avgBlankBuffer;
853  corrBlankValue = corrBlankValue/blankFwMap[i][j];
854  /* Corrected_MAX_Val = ((rawMAX-AverageOf(BLANK_buffer) / SloapRatio) / frehsweight */
855  var corrMaxValue = maxRawDataMap[i][j]-avgMaxBuffer;
856  corrMaxValue = corrMaxValue/slopeRatio;
857  corrMaxValue = corrMaxValue/maxFwMap[i][j];
858  /* Final_Val = ((Corrected_MAX_Val - Corrected_BLANK_Val) / (Corrected_MAX_Control - Corrected_BLANK_Control)) * Activation */
859  var procValue = ((corrMaxValue-corrBlankValue)/(corrMaxControl-corrBlankControl))*activity;
860  processDataMap[i][j]=procValue;
861  }
862  }
863  }
